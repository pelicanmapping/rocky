413  template<class T>
414  inline void ECS::VSG_SystemHelper<T>::record(vsg::RecordTraversal& rt) const
415  {
416  ROCKY_PROFILE_FUNCTION();
417 
418  const vsg::dmat4 identity_matrix = vsg::dmat4(1.0);
419 
420  // Get an optimized view of all this system's components:
421  auto view = registry.view<T>();
422 
423  using Entry = struct {
424  const T& component;
425  entt::entity entity;
426  };
427 
428  // Sort components into render sets by pipeline.
429  // If this system doesn't support multiple pipelines, just
430  // store them all together
431  std::vector<std::vector<Entry>> render_set(!pipelines.empty() ? pipelines.size() : 1);
432 
433  view.each([&](const entt::entity entity, const T& component)
434  {
435  // Is the component visible?
436  if (*component.active_ptr)
437  {
438  // Does it have a VSG node? If so, queue it up under the
439  // appropriate pipeline.
440  if (component.node)
441  {
442  auto& rs = !pipelines.empty() ? render_set[component.featureMask()] : render_set[0];
443  rs.emplace_back(Entry{ component, entity });
444  }
445 
446  if (!component.node || component.nodeDirty)
447  {
448  entities_to_initialize.push_back(entity);
449  }
450  }
451  });
452 
453  // Time to record all visible components.
454  // For each pipeline:
455  for (int p = 0; p < render_set.size(); ++p)
456  {
457  if (!render_set[p].empty())
458  {
459  // Bind the Graphics Pipeline for this render set, if there is one:
460  if (!pipelines.empty())
461  {
462  pipelines[p].commands->accept(rt);
463  }
464 
465  // Them record each component.
466  // If the component has a transform apply it too.
467  for (auto& e : render_set[p])
468  {
469  auto* xform = registry.try_get<Transform>(e.entity);
470  if (xform)
471  {
472  if (xform->push(rt, identity_matrix))
473  {
474  e.component.node->accept(rt);
475  xform->pop(rt);
476  }
477  }
478  else
479  {
480  e.component.node->accept(rt);
481  }
482  }
483  }
484  }

487  template<class T>
488  inline void ECS::VSG_SystemHelper<T>::initializeNewComponents(Runtime& runtime)
489  {
490  ROCKY_PROFILE_FUNCTION();
491 
492  // Components with VSG elements need to create and compile those
493  // elements before we can render them. These get put on the
494  // initialization list by the record traversal.
495  if (!entities_to_initialize.empty())
496  {
497  NodeComponent::Params params;
498  params.readerWriterOptions = runtime.readerWriterOptions;
499  params.sharedObjects = runtime.sharedObjects;
500 
501  for (auto& entity : entities_to_initialize)
502  {
503  auto& component = registry.get<T>(entity);
504 
505  // If it's marked dirty, dispose of it properly
506  if (component.node && component.nodeDirty)
507  {
508  runtime.dispose(component.node);
509  component.node = nullptr;
510  }
511 
512  if (!component.node)
513  {
514  // if we're using pipelines, find the one matching this
515  // component's feature set:
516  if (pipelines.empty())
517  params.layout = { };
518  else
519  params.layout = pipelines[component.featureMask()].config->layout;
520 
521  // Tell the component to create its VSG node(s)
522  component.initializeNode(params);
523 
524  // TODO: Replace this will some error checking
525  ROCKY_SOFT_ASSERT(component.node);
526  }
527 
528  // compile the vulkan objects
529  if (component.node)
530  {
531  runtime.compile(component.node);
532  }
533 
534  component.nodeDirty = false;
535  }
536 
537  // reset the list for the next frame
538  entities_to_initialize.clear();
539  }
