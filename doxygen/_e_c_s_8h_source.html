402  template<class T>
403  inline void ECS::SystemNode<T>::traverse(vsg::RecordTraversal& rt) const
404  {
405  const vsg::dmat4 identity_matrix = vsg::dmat4(1.0);
406 
407  // Sort components into render sets by pipeline. If this system doesn't support
408  // multiple pipelines, just store them all together in renderSet[0].
409  if (renderSet.empty())
410  renderSet.resize(!pipelines.empty() ? pipelines.size() : 1);
411 
412  // Get an optimized view of all this system's components:
413  registry.view<T, Visibility>().each([&](const entt::entity entity, const T& component, auto& visibility)
414  {
415  auto& renderable = registry.get<Renderable>(component.entity);
416  if (renderable.node)
417  {
418  auto& rs = !pipelines.empty() ? renderSet[featureMask(component)] : renderSet[0];
419  auto* transform = registry.try_get<Transform>(entity);
420  if (transform || visibility.visible)
421  rs.emplace_back(RenderLeaf{ renderable, transform, visibility.visible });
422  }
423 
424  if (!renderable.node || (renderable.revision != component.revision))
425  {
426  entities_to_update.push_back(entity);
427  }
428  });
429 
430  // Time to record all visible components.
431  // For each pipeline:
432  for (int p = 0; p < renderSet.size(); ++p)
433  {
434  if (!renderSet[p].empty())
435  {
436  // Bind the Graphics Pipeline for this render set, if there is one:
437  if (!pipelines.empty())
438  {
439  pipelines[p].commands->accept(rt);
440  }
441 
442  // Them record each component. If the component has a transform apply it too.
443  for (auto& leaf : renderSet[p])
444  {
445  if (leaf.transform)
446  {
447  if (leaf.transform->push(rt, identity_matrix))
448  {
449  if (leaf.visible)
450  leaf.renderable.node->accept(rt);
451 
452  leaf.transform->pop(rt);
453  }
454  }
455  else
456  {
457  if (leaf.visible)
458  leaf.renderable.node->accept(rt);
459  }
460  }
461 
462  // clear out for next time around.
463  renderSet[p].clear();
464  }
465  }
466  }
