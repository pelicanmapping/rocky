469  template<class T>
470  inline void ecs::SystemNode<T>::traverse(vsg::RecordTraversal& rt) const
471  {
472  const vsg::dmat4 identity_matrix = vsg::dmat4(1.0);
473  auto viewID = rt.getState()->_commandBuffer->viewID;
474 
475  // Sort components into render sets by pipeline. If this system doesn't support
476  // multiple pipelines, just store them all together in renderSet[0].
477  if (renderSet.empty())
478  renderSet.resize(!pipelines.empty() ? pipelines.size() : 1);
479 
480  // Get an optimized view of all this system's components:
481  registry.view<T, Visibility>().each([&](const entt::entity entity, const T& component, auto& visibility)
482  {
483  auto& renderable = registry.get<Renderable>(component.attach_point);
484  if (renderable.node)
485  {
486  auto& rs = !pipelines.empty() ? renderSet[featureMask(component)] : renderSet[0];
487  auto* transform = registry.try_get<Transform>(entity);
488  if (transform || visible(visibility, viewID))
489  {
490  rs.emplace_back(RenderLeaf{ renderable, transform, visible(visibility, viewID) });
491  }
492  }
493 
494  if (renderable.staged->has_value())
495  {
496  entities_to_update.emplace_back(entity);
497  }
498 
499  else if (renderable.revision != component.revision)
500  {
501  entities_to_update.emplace_back(entity);
502  renderable.revision = component.revision;
503  }
504  });
505 
506  // Time to record all visible components.
507  // For each pipeline:
508  for (int p = 0; p < renderSet.size(); ++p)
509  {
510  if (!renderSet[p].empty())
511  {
512  // Bind the Graphics Pipeline for this render set, if there is one:
513  if (!pipelines.empty())
514  {
515  pipelines[p].commands->accept(rt);
516  }
517 
518  // Them record each component. If the component has a transform apply it too.
519  for (auto& leaf : renderSet[p])
520  {
521  if (leaf.transform)
522  {
523  if (leaf.transform->push(rt, identity_matrix))
524  {
525  if (leaf.visible)
526  leaf.renderable.node->accept(rt);
527 
528  leaf.transform->pop(rt);
529  }
530  }
531  else
532  {
533  if (leaf.visible)
534  leaf.renderable.node->accept(rt);
535  }
536  }
537 
538  // clear out for next time around.
539  renderSet[p].clear();
540  }
541  }

545  template<class T>
546  inline void ecs::SystemNode<T>::update(Runtime& runtime)
547  {
548  std::vector<entt::entity> entities_to_create;
549 
550  for (auto& entity : entities_to_update)
551  {
552  T& component = registry.get<T>(entity);
553  Renderable& renderable = registry.get<Renderable>(component.attach_point);
554 
555  // if a staged node exists, swap it in.
556  vsg::ref_ptr<vsg::Node> new_node;
557  if (renderable.staged->get_and_clear(new_node))
558  {
559  if (renderable.node)
560  runtime.dispose(renderable.node);
561 
562  renderable.node = new_node;
563  }
564  else
565  {
566  // either the node doesn't exist yet, or the revision changed.
567  // Queue it up for creation.
568  entities_to_create.emplace_back(entity);
569  }
570  }
571 
572  // If we detected any entities that need new nodes, create them now.
573  if (!entities_to_create.empty())
574  {
575  if (manager)
576  {
577  bool ok = manager->entityCompileJobs.emplace(EntityCompileBatch{
578  std::move(entities_to_create),
579  this,
580  &runtime });
581 
582  if (!ok)
583  {
584  Log()->warn("Failed to enqueue entity compile job - queue overflow");
585  }
586  }
587  }
588 
589  entities_to_update.clear();
590  }
