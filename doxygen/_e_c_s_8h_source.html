358  template<class T, class R>
359  inline void ECS::SystemNode<T,R>::traverse(vsg::RecordTraversal& rt) const
360  {
361  const vsg::dmat4 identity_matrix = vsg::dmat4(1.0);
362 
363  using Entry = struct
364  {
365  const R& renderable;
366  entt::entity entity;
367  };
368 
369  // Sort components into render sets by pipeline.
370  // If this system doesn't support multiple pipelines, just
371  // store them all together
372  std::vector<std::vector<Entry>> render_set(!pipelines.empty() ? pipelines.size() : 1);
373 
374  // Get an optimized view of all this system's components:
375  registry.view<T, R>().each([&](const entt::entity entity, const T& component, const R& renderable)
376  {
377  // Is the component visible?
378  if (*component.active_ptr)
379  {
380  // Does it have a VSG node? If so, queue it up under the
381  // appropriate pipeline.
382  if (renderable.node)
383  {
384  auto& rs = !pipelines.empty() ? render_set[featureMask(component)] : render_set[0];
385  rs.emplace_back(Entry{ renderable, entity });
386  }
387 
388  if (!renderable.node || (renderable.revision != component.revision))
389  {
390  entities_to_update.push_back(entity);
391  }
392  }
393  });
394 
395  // Time to record all visible components.
396  // For each pipeline:
397  for (int p = 0; p < render_set.size(); ++p)
398  {
399  if (!render_set[p].empty())
400  {
401  // Bind the Graphics Pipeline for this render set, if there is one:
402  if (!pipelines.empty())
403  {
404  pipelines[p].commands->accept(rt);
405  }
406 
407  // Them record each component.
408  // If the component has a transform apply it too.
409  for (auto& e : render_set[p])
410  {
411  auto* xform = registry.try_get<Transform>(e.entity);
412  if (xform)
413  {
414  if (xform->push(rt, identity_matrix))
415  {
416  e.renderable.node->accept(rt);
417  xform->pop(rt);
418  }
419  }
420  else
421  {
422  e.renderable.node->accept(rt);
423  }
424  }
425  }
426  }
427  }
