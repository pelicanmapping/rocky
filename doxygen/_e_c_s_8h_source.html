423  template<class T>
424  inline void ECS::SystemNode<T>::traverse(vsg::RecordTraversal& rt) const
425  {
426  const vsg::dmat4 identity_matrix = vsg::dmat4(1.0);
427 
428  // Sort components into render sets by pipeline. If this system doesn't support
429  // multiple pipelines, just store them all together in renderSet[0].
430  if (renderSet.empty())
431  renderSet.resize(!pipelines.empty() ? pipelines.size() : 1);
432 
433  // Get an optimized view of all this system's components:
434  registry.view<T>().each([&](const entt::entity entity, const T& component)
435  {
436  // Is the component visible?
437  if (*component.visible_ptr)
438  {
439  auto& renderable = registry.get<Renderable>(component.entity);
440  if (renderable.node)
441  {
442  auto& rs = !pipelines.empty() ? renderSet[featureMask(component)] : renderSet[0];
443  rs.emplace_back(RenderLeaf{ renderable, entity });
444  }
445 
446  if (!renderable.node || (renderable.revision != component.revision))
447  {
448  entities_to_update.push_back(entity);
449  }
450  }
451  });
452 
453  // Time to record all visible components.
454  // For each pipeline:
455  for (int p = 0; p < renderSet.size(); ++p)
456  {
457  if (!renderSet[p].empty())
458  {
459  // Bind the Graphics Pipeline for this render set, if there is one:
460  if (!pipelines.empty())
461  {
462  pipelines[p].commands->accept(rt);
463  }
464 
465  // Them record each component. If the component has a transform apply it too.
466  for (auto& leaf : renderSet[p])
467  {
468  auto* xform = registry.try_get<Transform>(leaf.entity);
469  if (xform)
470  {
471  if (xform->push(rt, identity_matrix))
472  {
473  leaf.renderable.node->accept(rt);
474  xform->pop(rt);
475  }
476  }
477  else
478  {
479  leaf.renderable.node->accept(rt);
480  }
481  }
482 
483  // clear out for next time around.
484  renderSet[p].clear();
485  }
486  }
487  }
