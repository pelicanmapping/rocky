373  template<class COMPONENT>
374  inline void ECS::SystemNodeHelper<COMPONENT>::record(vsg::RecordTraversal& rt) const
375  {
376  const vsg::dmat4 identity_matrix = vsg::dmat4(1.0);
377 
378  // Get an optimized view of all this system's components:
379  auto view = registry.view<COMPONENT>();
380 
381  using Entry = struct {
382  const COMPONENT& component;
383  entt::entity entity;
384  };
385 
386  // Sort components into render sets by pipeline.
387  // If this system doesn't support multiple pipelines, just
388  // store them all together
389  std::vector<std::vector<Entry>> render_set(!pipelines.empty() ? pipelines.size() : 1);
390 
391  view.each([&](const entt::entity entity, const COMPONENT& component)
392  {
393  // Is the component visible?
394  if (*component.active_ptr)
395  {
396  // Does it have a VSG node? If so, queue it up under the
397  // appropriate pipeline.
398  if (component.node)
399  {
400  auto& rs = !pipelines.empty() ? render_set[component.featureMask()] : render_set[0];
401  rs.emplace_back(Entry{ component, entity });
402  }
403 
404  if (!component.node || component.nodeDirty)
405  {
406  entities_to_initialize.push_back(entity);
407  }
408  }
409  });
410 
411  // Time to record all visible components.
412  // For each pipeline:
413  for (int p = 0; p < render_set.size(); ++p)
414  {
415  if (!render_set[p].empty())
416  {
417  // Bind the Graphics Pipeline for this render set, if there is one:
418  if (!pipelines.empty())
419  {
420  pipelines[p].commands->accept(rt);
421  }
422 
423  // Them record each component.
424  // If the component has a transform apply it too.
425  for (auto& e : render_set[p])
426  {
427  auto* xform = registry.try_get<Transform>(e.entity);
428  if (xform)
429  {
430  if (xform->push(rt, identity_matrix))
431  {
432  e.component.node->accept(rt);
433  xform->pop(rt);
434  }
435  }
436  else
437  {
438  e.component.node->accept(rt);
439  }
440  }
441  }
442  }

445  template<class COMPONENT>
446  inline void ECS::SystemNodeHelper<COMPONENT>::initializeNewComponents(Runtime& runtime)
447  {
448  // Components with VSG elements need to create and compile those
449  // elements before we can render them. These get put on the
450  // initialization list by the record traversal.
451  if (!entities_to_initialize.empty())
452  {
453  SystemNode::InitContext context
454  {
455  {}, // will fill this in below
456  runtime.readerWriterOptions,
457  runtime.sharedObjects
458  };
459 
460  for (auto& entity : entities_to_initialize)
461  {
462  auto& component = registry.get<COMPONENT>(entity);
463 
464  // If it's marked dirty, dispose of it properly
465  if (component.node && component.nodeDirty)
466  {
467  runtime.dispose(component.node);
468  component.node = nullptr;
469  }
470 
471  if (!component.node)
472  {
473  // if we're using pipelines, find the one matching this
474  // component's feature set:
475  if (pipelines.empty())
476  context.layout = { };
477  else
478  context.layout = pipelines[component.featureMask()].config->layout;
479 
480  // Tell the component to create its VSG node(s)
481  if (initializeComponent)
482  {
483  initializeComponent(component, context);
484  }
485 
486  // TODO: Replace this will some error checking
487  ROCKY_SOFT_ASSERT(component.node);
488  }
489 
490  // compile the vulkan objects
491  if (component.node)
492  {
493  runtime.compile(component.node);
494  }
495 
496  component.nodeDirty = false;
497  }
498 
499  // reset the list for the next frame
500  entities_to_initialize.clear();
501  }
