38 template <typename T>
39 inline std::size_t hash_value_unsigned(T val)
40 {
41 const int size_t_bits = std::numeric_limits<std::size_t>::digits;
42 const int length = (std::numeric_limits<T>::digits - 1) / size_t_bits;
43 std::size_t seed = 0;
44 for (unsigned int i = length * size_t_bits; i > 0; i -= size_t_bits)
45 seed ^= (std::size_t)(val >> i) + (seed << 6) + (seed >> 2);
46 seed ^= (std::size_t)val + (seed << 6) + (seed >> 2);
47 return seed;
48 }

131 // true if 2 segments intersect; intersection point in "out"
132 bool intersect(const segment_t& rhs, vert_t& out, vert_t::value_type& u) const
133 {
134 vert_t r = second - first;
135 vert_t s = rhs.second - rhs.first;
136 vert_t::value_type det = r.cross2d(s);
137
138 if (equivalent(det, zero, 1e-5))
139 return false;
140
141 vert_t diff = rhs.first - first;
142 u = diff.cross2d(s) / det;
143 vert_t::value_type v = diff.cross2d(r) / det;
144
145 out = first + (r * u);
146 return (u > 0.0 && u < 1.0 && v > 0.0 && v < 1.0);
147 }

186 // true if point P is one of the triangle's verts
187 inline bool is_vertex(const vert_t& p, vert_t::value_type epsilon) const
188 {
189 if (equivalent(p.x, p0.x, epsilon) &&
190 equivalent(p.y, p0.y, epsilon))
191 return true;
192 if (equivalent(p.x, p1.x, epsilon) &&
193 equivalent(p.y, p1.y, epsilon))
194 return true;

196 equivalent(p.y, p2.y, epsilon))
197 return true;

199 return false;

202 inline int get_vertex(const vert_t& p, vert_t::value_type epsilon) const
203 {

205 equivalent(p.y, p0.y, epsilon))
206 return i0;
207 if (equivalent(p.x, p1.x, epsilon) &&
208 equivalent(p.y, p1.y, epsilon))
209 return i1;
210 if (equivalent(p.x, p2.x, epsilon) &&
211 equivalent(p.y, p2.y, epsilon))

217 inline bool get_barycentric(const vert_t& p, vert_t& out, vert_t::value_type epsilon) const
218 {
219 vert_t v0 = p1 - p0, v1 = p2 - p0, v2 = p - p0;
220 vert_t::value_type d00 = v0.dot2d(v0);
221 vert_t::value_type d01 = v0.dot2d(v1);
222 vert_t::value_type d11 = v1.dot2d(v1);
223 vert_t::value_type d20 = v2.dot2d(v0);
224 vert_t::value_type d21 = v2.dot2d(v1);
225 vert_t::value_type denom = d00 * d11 - d01 * d01;
226
227 // means that one of more of the triangles points are coincident:
228 if (equivalent(denom, 0.0, epsilon))
229 return false;
230
231 out.y = (d11 * d20 - d01 * d21) / denom;
232 out.z = (d00 * d21 - d01 * d20) / denom;
233 out.x = 1.0 - out.y - out.z;
234
235 return true;
236 }

312 // add new triangle to the mesh from 3 indices
313 UID add_triangle(int i0, int i1, int i2)
314 {
315 if (i0 == i1 || i1 == i2 || i2 == i0)
316 return -1;
317
318 UID uid(uidgen++);
319 triangle_t tri;
320 tri.uid = uid;
321 tri.i0 = i0;
322 tri.i1 = i1;
323 tri.i2 = i2;
324 tri.p0 = get_vertex(i0);
325 tri.p1 = get_vertex(i1);
326 tri.p2 = get_vertex(i2);
327 tri.a_min[0] = std::min(tri.p0.x, std::min(tri.p1.x, tri.p2.x));
328 tri.a_min[1] = std::min(tri.p0.y, std::min(tri.p1.y, tri.p2.y));
329 tri.a_max[0] = std::max(tri.p0.x, std::max(tri.p1.x, tri.p2.x));
330 tri.a_max[1] = std::max(tri.p0.y, std::max(tri.p1.y, tri.p2.y));
331 tri.centroid = (tri.p0 + tri.p1 + tri.p2) * one_third;
332
333 // "2d_degenerate" means that either a) at least 2 points are coincident, or
334 // b) at least two edges are basically coincident (in the XY plane)
335 tri.is_2d_degenerate =
336 same_vert(tri.p0, tri.p1, epsilon) ||
337 same_vert(tri.p1, tri.p2, epsilon) ||
338 same_vert(tri.p2, tri.p0, epsilon) ||
339 same_vert((tri.p1 - tri.p0).normalize2d(), (tri.p2 - tri.p0).normalize2d(), epsilon) ||
340 same_vert((tri.p2 - tri.p1).normalize2d(), (tri.p0 - tri.p1).normalize2d(), epsilon) ||
341 same_vert((tri.p0 - tri.p2).normalize2d(), (tri.p1 - tri.p2).normalize2d(), epsilon);
342
343 triangles.emplace(uid, tri);
344 _spatial_index.Insert(tri.a_min, tri.a_max, uid);
345
346 ++_num_edits;
347
348 return uid;
349 }

375 // Add a new vertex (or lookup a matching one) and return its index.
376 // If the vertex already exists, update its marker if necessary.
377 int get_or_create_vertex(const vert_t& input, int marker)
378 {
379 int index;
380 vert_table_t::iterator i = _vert_lut.find(input);
381 if (i != _vert_lut.end())
382 {
383 index = i->second;
384 markers[i->second] |= marker;
385 }
386 else if (verts.size() + 1 < 0xFFFF)
387 {
388 verts.push_back(input);
389 markers.push_back(marker);
390 _vert_lut[input] = (int)verts.size() - 1;
391 index = (int)verts.size() - 1;
392 }
393 else
394 {
395 return -1;
396 }
397
398 return index;
399 }

418 // insert a point into the mesh, cutting triangles as necessary
419 void insert(const vert_t& vert, int marker)
420 {
421 // search for possible intersecting triangles (should only be one)
422 vert_t::value_type a_min[2];
423 vert_t::value_type a_max[2];
424 a_min[0] = a_max[0] = vert.x;
425 a_min[1] = a_max[1] = vert.y;
426
427 std::vector<UID> uids;
428
429 _spatial_index.Search(
430 a_min, a_max,
431 [&uids](const UID& u) {
432 uids.push_back(u);
433 return true;
434 });
435
436 for (auto uid : uids)
437 {
438 triangle_t& tri = triangles[uid];
439
440 if (tri.is_2d_degenerate)
441 continue;
442
443 if (tri.contains_2d(vert, epsilon))
444 {
445 inside_split(tri, vert, nullptr, marker);
446 // dont't break -- could split two triangles if on edge.
447 //break;
448 }
449 }
450 }

452 // insert a segment into the mesh, cutting triangles as necessary
453 void insert(const segment_t& seg, int marker)
454 {
455 // search for possible intersecting triangles:
456 vert_t::value_type a_min[2];
457 vert_t::value_type a_max[2];
458 a_min[0] = std::min(seg.first.x, seg.second.x);
459 a_min[1] = std::min(seg.first.y, seg.second.y);
460 a_max[0] = std::max(seg.first.x, seg.second.x);
461 a_max[1] = std::max(seg.first.y, seg.second.y);
462 std::vector<UID> uids;
463
464 _spatial_index.Search(
465 a_min, a_max,
466 [&uids](const UID& u) {
467 uids.push_back(u);
468 return true;
469 });
470
471 // The working set of triangles which we will add to if we have
472 // to split triangles. Any triangle only needs to be split once,
473 // even if it gets intersected multiple times. That is because each
474 // split generates new traigles, and discards the original, and further
475 // splits will just happen on the new triangles later. (That's why
476 // every split operation is followed by a "continue" to short-circuit
477 // to loop)
478 std::list<UID> uid_list;
479 std::copy(uids.begin(), uids.end(), std::back_inserter(uid_list));
480 for (auto uid : uid_list)
481 {
482 triangle_t& tri = triangles[uid];
483
484 // skip triangles that are "degenerate" in 2D. We will keep them
485 // because they may NOT be degenerate in 3D (e.g. steep slopes).
486 if (tri.is_2d_degenerate)
487 continue;
488
489 // first see if one of the endpoints falls within a triangle.
490 // if so, split the triangle into 2 or 3 new ones, and mark
491 // all of its verts as CONSTRAINT so they will not be subject
492 // to morphing.
493 if (tri.contains_2d(seg.first, epsilon))
494 {
495 if (inside_split(tri, seg.first, &uid_list, marker))
496 continue;
497 }
498
499 if (tri.contains_2d(seg.second, epsilon))
500 {
501 if (inside_split(tri, seg.second, &uid_list, marker))
502 continue;
503 }
504
505 // Next try to intersect the segment with a triangle edge.
506 // In each case, make sure the intersection point isn't
507 // coincident with an existing vertex (in which case ignore it).
508 // If we do split an edge, mark all of its verts as CONSTRAINT
509 // so they will not be subject to morphing; furthermore, convery
510 // any BOUNDARY markers to the new vert so we can maintain boundaries
511 // for skirt generation.
512 vert_t out;
513 vert_t::value_type u;
514 UID new_uid;
515 int new_i;
516
517 // does the segment cross first triangle edge?
518 segment_t edge0(tri.p0, tri.p1);
519 if (seg.intersect(edge0, out, u) && !tri.is_vertex(out, epsilon))
520 {
521 int new_marker = marker;
522 if ((markers[tri.i0] & _boundary_marker) && (markers[tri.i1] & _boundary_marker))
523 new_marker |= _boundary_marker;
524
525 new_i = get_or_create_vertex(out, new_marker);
526 if (new_i < 0)
527 return;
528
529 int new_tris = 0;
530
531 new_uid = add_triangle(new_i, tri.i2, tri.i0);
532 if (new_uid >= 0) {
533 markers[tri.i2] |= _constraint_marker;
534 markers[tri.i0] |= _constraint_marker;
535 uid_list.push_back(new_uid);
536 ++new_tris;
537 }
538
539 new_uid = add_triangle(new_i, tri.i1, tri.i2);
540 if (new_uid >= 0) {
541 markers[tri.i1] |= _constraint_marker;
542 markers[tri.i2] |= _constraint_marker;
543 uid_list.push_back(new_uid);
544 ++new_tris;
545 }
546
547 if (new_tris > 0)
548 {
549 if (marker_not_set(new_i, _has_elevation_marker) &&
550 marker_is_set(tri.i0, _has_elevation_marker) && marker_is_set(tri.i1, _has_elevation_marker))
551 {
552 auto z0 = get_vertex(tri.i0).z, z1 = get_vertex(tri.i1).z;
553 get_vertex(new_i).z = z0 + u * (z1 - z0);
554 set_marker(new_i, _has_elevation_marker);
555 }
556
557 remove_triangle(tri);
558 continue;
559 }
560 }
561
562 // does the segment cross second triangle edge?
563 segment_t edge1(tri.p1, tri.p2);
564 if (seg.intersect(edge1, out, u) && !tri.is_vertex(out, epsilon))
565 {
566 int new_marker = marker;
567 if ((markers[tri.i1] & _boundary_marker) && (markers[tri.i2] & _boundary_marker))
568 new_marker |= _boundary_marker;
569
570 new_i = get_or_create_vertex(out, new_marker);
571 if (new_i < 0)
572 return;
573
574 int new_tris = 0;
575
576 new_uid = add_triangle(new_i, tri.i0, tri.i1);
577 if (new_uid >= 0) {
578 markers[tri.i0] |= _constraint_marker;
579 markers[tri.i1] |= _constraint_marker;
580 uid_list.push_back(new_uid);
581 ++new_tris;
582 }
583
584 new_uid = add_triangle(new_i, tri.i2, tri.i0);
585 if (new_uid >= 0) {
586 markers[tri.i2] |= _constraint_marker;
587 markers[tri.i0] |= _constraint_marker;
588 uid_list.push_back(new_uid);
589 ++new_tris;
590 }
591
592 if (new_tris > 0)
593 {
594 if (marker_not_set(new_i, _has_elevation_marker) &&
595 marker_is_set(tri.i1, _has_elevation_marker) && marker_is_set(tri.i2, _has_elevation_marker))
596 {
597 auto z0 = get_vertex(tri.i1).z, z1 = get_vertex(tri.i2).z;
598 get_vertex(new_i).z = z0 + u * (z1 - z0);
599 set_marker(new_i, _has_elevation_marker);
600 }
601
602 remove_triangle(tri);
603 continue;
604 }
605 }
606
607 // does the segment cross third triangle edge?
608 segment_t edge2(tri.p2, tri.p0);
609 if (seg.intersect(edge2, out, u) && !tri.is_vertex(out, epsilon))
610 {
611 int new_marker = marker;
612 if ((markers[tri.i2] & _boundary_marker) && (markers[tri.i0] & _boundary_marker))
613 new_marker |= _boundary_marker;
614
615 new_i = get_or_create_vertex(out, new_marker);
616 if (new_i < 0)
617 return;
618
619 int new_tris = 0;
620
621 new_uid = add_triangle(new_i, tri.i1, tri.i2);
622 if (new_uid >= 0) {
623 markers[tri.i1] |= _constraint_marker;
624 markers[tri.i2] |= _constraint_marker;
625 uid_list.push_back(new_uid);
626 ++new_tris;
627 }
628
629 new_uid = add_triangle(new_i, tri.i0, tri.i1);
630 if (new_uid >= 0) {
631 markers[tri.i0] |= _constraint_marker;
632 markers[tri.i1] |= _constraint_marker;
633 uid_list.push_back(new_uid);
634 ++new_tris;
635 }
636
637 if (new_tris > 0)
638 {
639 if (marker_not_set(new_i, _has_elevation_marker) &&
640 marker_is_set(tri.i2, _has_elevation_marker) && marker_is_set(tri.i0, _has_elevation_marker))
641 {
642 auto z0 = get_vertex(tri.i2).z, z1 = get_vertex(tri.i0).z;
643 get_vertex(new_i).z = z0 + u * (z1 - z0);
644 set_marker(new_i, _has_elevation_marker);
645 }
646
647 remove_triangle(tri);
648 continue;
649 }
650 }
651 }
652 }

654 // inserts point "p" into the interior of triangle "tri",
655 // adds three new triangles, and removes the original triangle.
656 // return true if a split actual happened
657 bool inside_split(triangle_t& tri, const vert_t& p, std::list<UID>* uid_list, int new_marker)
658 {
659 int new_i = get_or_create_vertex(p, new_marker);
660 if (new_i < 0)
661 return false;
662
663 // if this vertex is already one of the triangle's verts, no split is necessary
664 // and we are done.
665 if (tri.is_vertex(new_i))
666 return false;
667
668 UID new_uid;
669 int new_tris = 0;
670
671 // calculate the barycentric coordinates of the new point
672 // within the target triangle so we can detect any split
673 // that occurs right on an edge and then bypass creating
674 // a degenerate triangle.
675 vert_t bary(1, 1, 1);
676 if (tri.get_barycentric(p, bary, epsilon) == false)
677 return false;
678
679 if (!equivalent(bary[2], 0.0, epsilon)) {
680 new_uid = add_triangle(tri.i0, tri.i1, new_i);
681 if (new_uid >= 0 && uid_list) {
682 markers[tri.i0] |= _constraint_marker;
683 markers[tri.i1] |= _constraint_marker;
684 uid_list->push_back(new_uid);
685 ++new_tris;
686 }
687 }
688
689 if (!equivalent(bary[0], 0.0, epsilon)) {
690 new_uid = add_triangle(tri.i1, tri.i2, new_i);
691 if (new_uid >= 0 && uid_list) {
692 markers[tri.i1] |= _constraint_marker;
693 markers[tri.i2] |= _constraint_marker;
694 uid_list->push_back(new_uid);
695 ++new_tris;
696 }
697 }
698
699 if (!equivalent(bary[1], 0.0, epsilon)) {
700 new_uid = add_triangle(tri.i2, tri.i0, new_i);
701 if (new_uid >= 0 && uid_list) {
702 markers[tri.i2] |= _constraint_marker;
703 markers[tri.i0] |= _constraint_marker;
704 uid_list->push_back(new_uid);
705 ++new_tris;
706 }
707 }
708
709 if (new_tris > 0)
710 {
711 // if all verts in the surrounding triangle have elevation data,
712 // but this new one doesn't, set it via interpolation.
713 if (marker_not_set(new_i, _has_elevation_marker) &&
714 marker_is_set(tri.i0, _has_elevation_marker) &&
715 marker_is_set(tri.i1, _has_elevation_marker) &&
716 marker_is_set(tri.i2, _has_elevation_marker))
717 {
718 // interpolate the Z value from the triangle:
719 get_vertex(new_i).z =
720 get_vertex(tri.i0).z * bary[0] +
721 get_vertex(tri.i1).z * bary[1] +
722 get_vertex(tri.i2).z * bary[2];
723
724 set_marker(new_i, _has_elevation_marker);
725 }
726
727 remove_triangle(tri);
728 }
729
730 return (new_tris > 0);
731 }
732
733 vert_t point_on_edge_closest_to(const edge_t& edge, const vert_t& p) const {
734 const vert_t& e1 = get_vertex(edge._i0);
735 const vert_t& e2 = get_vertex(edge._i1);
736 vert_t qp = e2 - e1;
737 vert_t xp = p - e1;
738 double u = xp.dot2d(qp) / qp.dot2d(qp);
739 if (u < 0.0) return e1;
740 else if (u > 1.0) return e2;
741 else return e1 + qp * u;
742 }
743 };

771 void add_triangle(const triangle_t& tri, const mesh_t& mesh, int marker_mask)
772 {
773 bool m0 = (mesh.markers[tri.i0] & marker_mask) != 0;
774 bool m1 = (mesh.markers[tri.i1] & marker_mask) != 0;
775 bool m2 = (mesh.markers[tri.i2] & marker_mask) != 0;
776
777 if (m0 && m1)
778 _edges.emplace(tri.i0, tri.i1);
779 if (m1 && m2)
780 _edges.emplace(tri.i1, tri.i2);
781 if (m2 && m0)
782 _edges.emplace(tri.i2, tri.i0);
783 }

785 bool point_on_any_edge_closest_to(const vert_t& p, const mesh_t& mesh, vert_t& closest) const
786 {
787 double min_distance2 = DBL_MAX;
788
789 for (auto& edge : _edges)
790 {
791 vert_t c = mesh.point_on_edge_closest_to(edge, p);
792 auto d2 = (c - p).length2d_squared();
793 if (d2 < min_distance2)
794 {
795 min_distance2 = d2;
796 closest = c;
797 }
798 }
799 return min_distance2 < DBL_MAX;
800 }

897 // attempt to find the concave hull by walking the outside of subgraph.
898 void get_hull(int graphid, const mesh_t& mesh, std::vector<int>& hull) const
899 {
900 const node_t* start = nullptr;
901 vert_t::value_type min_y = DBL_MAX;
902 vert_t::value_type min_x = DBL_MAX;
903
904 // find the vertex with minimum y in graph graphid.
905 // TODO: it would be faster to find a vertex in graph graphid,
906 // and then traverse from there.
907 for (const auto& node_iter : _nodes)
908 {
909 const node_t& node = node_iter.second;
910 if (node._graphid == graphid)
911 {
912 const vert_t& vert = mesh.get_vertex(node._vertex_index);
913 if (start == nullptr || vert.y < min_y)
914 {
915 start = &node;
916 min_y = vert.y;
917 }
918 }
919 }
920
921 if (start == nullptr)
922 return;
923
924 // next walk the boundary in a clockwise direction
925 const node_t* curr_node = start;
926 const vert_t& first = mesh.get_vertex(curr_node->_vertex_index);
927 const node_t* prev_node = nullptr;
928 vert_t prev_vert(first + vert_t(0, -1, 0));
929
930 std::set<const node_t*> visited;
931
932 while (true)
933 {
934 hull.push_back(curr_node->_vertex_index);
935
936 double best_score = DBL_MAX;
937 const node_t* best_edge = nullptr;
938 const vert_t& curr_vert = mesh.get_vertex(curr_node->_vertex_index);
939
940 vert_t invec = (curr_vert - prev_vert).normalize2d();
941
942 for (auto& edge : curr_node->_edges)
943 {
944 if (visited.find(edge) != visited.end())
945 continue;
946 //if (edge == prev_node) // no backtracking
947 // continue;
948
949 const vert_t& next_vert = mesh.get_vertex(edge->_vertex_index);
950 vert_t outvec = (next_vert - curr_vert).normalize2d();
951
952 double turn = invec.cross2d(outvec) > 0.0 ? -1.0 : 1.0;
953 double dot = 1.0 - (0.5 * (invec.dot2d(outvec) + 1.0));
954 double score = turn * dot;
955
956 if (score < best_score)
957 {
958 best_score = score;
959 best_edge = edge;
960 }
961 }
962
963 if (best_edge == nullptr)
964 {
965 break;
966 }
967
968 if (best_edge == start)
969 {
970 // done!
971 break;
972 }
973
974 if (curr_node != start)
975 visited.insert(curr_node);
976
977 prev_node = curr_node;
978 prev_vert = curr_vert;
979 curr_node = best_edge;
980 }
981 }
