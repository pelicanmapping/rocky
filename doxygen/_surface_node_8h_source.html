80  inline bool SurfaceNode::isVisible(vsg::RecordTraversal& rv) const
81  {
82  auto* state = rv.getState();
83 
84  // bounding box visibility check; this is much tighter than the bounding
85  // sphere. _frustumStack.top() contains the frustum in world coordinates.
86  // https://github.com/vsg-dev/VulkanSceneGraph/blob/master/include/vsg/vk/State.h#L267
87  // The first 8 points in _worldPoints are the 8 corners of the bounding box
88  // in world coordinates.
89  // Note: POLYTOPE_SIZE is defined in vsg plane.h
90  auto& frustum = state->_frustumStack.top();
91  int p;
92  for (int f = 0; f < POLYTOPE_SIZE; ++f) {
93  for (p = 0; p < 8; ++p)
94  if (vsg::distance(frustum.face[f], _worldPoints[p]) > 0.0) // visible?
95  break;
96  if (p == 8)
97  return false;
98  }
99 
100  // Horizon culling:
101  auto& viewlocal = _viewlocal[state->_commandBuffer->viewID];
102  if (!viewlocal.horizon)
103  {
104  rv.getValue("rocky.horizon", viewlocal.horizon);
105  }
106  if (viewlocal.horizon)
107  {
108  if (_horizonCullingPoint_valid)
109  {
110  return viewlocal.horizon->isVisible(_horizonCullingPoint);
111  }
112  else
113  {
114  for (p = 0; p < 4; ++p)
115  {
116  auto& wp = _worldPoints[p];
117  if (viewlocal.horizon->isVisible(wp.x, wp.y, wp.z))
118  return true;
119  }
120  return false;
121  }
122  }
123 
124  return true;
125  }
126 }
