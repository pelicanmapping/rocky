115  inline vsg::ref_ptr<vsg::Data> moveImageData(shared_ptr<Image> image)
116  {
117  if (!image) return { };
118 
119  switch (image->pixelFormat())
120  {
121  case Image::R8_UNORM:
122  return move<unsigned char>(image, VK_FORMAT_R8_UNORM);
123  break;
124  case Image::R8G8_UNORM:
125  return move<vsg::ubvec2>(image, VK_FORMAT_R8G8_UNORM);
126  break;
127  case Image::R8G8B8_UNORM:
128  return move<vsg::ubvec3>(image, VK_FORMAT_R8G8B8_UNORM);
129  break;
130  case Image::R8G8B8A8_UNORM:
131  return move<vsg::ubvec4>(image, VK_FORMAT_R8G8B8A8_UNORM);
132  break;
133  case Image::R16_UNORM:
134  return move<unsigned short>(image, VK_FORMAT_R16_UNORM);
135  break;
136  case Image::R32_SFLOAT:
137  return move<float>(image, VK_FORMAT_R32_SFLOAT);
138  break;
139  case Image::R64_SFLOAT:
140  return move<double>(image, VK_FORMAT_R64_SFLOAT);
141  break;
142  };
143 
144  return { };
145  }

163  // Convert a vsg::Data structure to an Image if possible
164  inline Result<shared_ptr<Image>> makeImageFromVSG(vsg::ref_ptr<vsg::Data> data)
165  {
166  if (!data)
167  return Status(Status::ResourceUnavailable, "Data is empty");
168 
169  // TODO: move this into a utility somewhere
170  auto vkformat = data->properties.format;
171 
172  Image::PixelFormat format =
173  vkformat == VK_FORMAT_R8_UNORM ? Image::R8_UNORM :
174  vkformat == VK_FORMAT_R8G8_UNORM ? Image::R8G8_UNORM :
175  vkformat == VK_FORMAT_R8G8B8_UNORM ? Image::R8G8B8_UNORM :
176  vkformat == VK_FORMAT_R8G8B8A8_UNORM ? Image::R8G8B8A8_UNORM :
177  vkformat == VK_FORMAT_R16_UNORM ? Image::R16_UNORM :
178  vkformat == VK_FORMAT_R32_SFLOAT ? Image::R32_SFLOAT :
179  vkformat == VK_FORMAT_R64_SFLOAT ? Image::R64_SFLOAT :
180  Image::UNDEFINED;
181 
182  if (format == Image::UNDEFINED)
183  {
184  return Status(Status::ResourceUnavailable, "Unsupported image format");
185  }
186 
187  auto image = Image::create(
188  format,
189  data->width(),
190  data->height(),
191  data->depth());
192 
193  memcpy(image->data<uint8_t>(), data->dataPointer(), image->sizeInBytes());
194 
195  if (data->properties.origin == vsg::TOP_LEFT)
196  {
197  image->flipVerticalInPlace();
198  }
199 
200  return Result(image);
201  }
