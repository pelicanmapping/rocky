566 void _dispatch_delegate(std::function<bool()>& delegate, const context& context)
567 {
568 if (!_done)
569 {
570 // If we have a group semaphore, acquire it BEFORE queuing the job
571 if (context.group)
572 {
573 context.group->acquire();
574 }
575
576 if (_target_concurrency > 0)
577 {
578 std::lock_guard<std::mutex> lock(_queue_mutex);
579
580 _queue.emplace_back(detail::job{ context, delegate });
581 _queue_size++;
582
583 _metrics.pending++;
584 _metrics.total++;
585 _block.notify_one();
586 }
587 else
588 {
589 // no threads? run synchronously.
590 delegate();
591
592 if (context.group)
593 {
594 context.group->release();
595 }
596 }

603 inline bool _take_job(detail::job& output, bool lock)
604 {
605 if (lock)
606 {
607 std::lock_guard<std::mutex> lock(_queue_mutex);
608 return _take_job(output, false);
609 }
610 else if (!_done && _queue_size > 0)
611 {
612 auto ptr = _queue.end();
613 float highest_priority = -FLT_MAX;
614 for (auto iter = _queue.begin(); iter != _queue.end(); ++iter)
615 {
616 float priority = iter->ctx.priority != nullptr ?
617 iter->ctx.priority() :
618 0.0f;
619
620 if (ptr == _queue.end() || priority > highest_priority)
621 {
622 ptr = iter;
623 highest_priority = priority;
624 }
625 }
626
627 if (ptr == _queue.end())
628 ptr = _queue.begin();
629
630 output = std::move(*ptr);
631 _queue.erase(ptr);
632 _queue_size--;
633 _metrics.pending--;
634 return true;
635 }
636 return false;

888 inline void jobpool::run()
889 {
890 while (!_done)
891 {
892 detail::job next;
893 bool have_next = false;
894 {
895 if (_can_steal_work && instance()._stealing_allowed)
896 {
897 {
898 std::unique_lock<std::mutex> lock(_queue_mutex);
899
900 // work-stealing enabled: wait until any queue is non-empty
901 _block.wait(lock, [this]() { return get_metrics()->total_pending() > 0 || _done; });
902
903 if (!_done && _queue_size > 0)
904 {
905 have_next = _take_job(next, false);
906 }
907 }
908
909 if (!_done && !have_next)
910 {
911 have_next = detail::steal_job(this, next);
912 }
913 }
914 else
915 {
916 std::unique_lock<std::mutex> lock(_queue_mutex);
917
918 // wait until just our local queue is non-empty
919 _block.wait(lock, [this] { return (_queue_size > 0) || _done; });
920
921 if (!_done && _queue_size > 0)
922 {
923 have_next = _take_job(next, false);
924 }
925 }
926 }
927
928 if (have_next)
929 {
930 _metrics.running++;
931
932 auto t0 = std::chrono::steady_clock::now();
933
934 bool job_executed = next._delegate();
935
936 auto duration = std::chrono::steady_clock::now() - t0;
937
938 if (job_executed == false)
939 {
940 _metrics.canceled++;
941 }
942
943 // release the group semaphore if necessary
944 if (next.ctx.group != nullptr)
945 {
946 next.ctx.group->release();
947 }
948
949 _metrics.running--;
950 }
951
952 // See if we no longer need this thread because the
953 // target concurrency has been reduced
954 std::lock_guard<std::mutex> lock(_quit_mutex);
955
956 if (_target_concurrency < _metrics.concurrency)
957 {
958 _metrics.concurrency--;
959 break;
960 }
961 }
962 }

985 inline void jobpool::stop_threads()
986 {
987 _done = true;
988
989 // Clear out the queue
990 std::lock_guard<std::mutex> lock(_queue_mutex);
991
992 // reset any group semaphores so that JobGroup.join()
993 // will not deadlock.
994 for (auto& queuedjob : _queue)
995 {
996 if (queuedjob.ctx.group != nullptr)
997 {
998 queuedjob.ctx.group->release();
999 }
1000 }
1001 _queue.clear();
1002 _queue_size = 0;
1003
1004 // wake up all threads so they can exit
1005 _block.notify_all();
1006 }

1052 template<typename T>
1053 template<typename F, typename R>
1054 inline future<R> future<T>::then_dispatch(F func, const context& con)
1055 {
1056 // The future result of F.
1057 // In this case, the continuation task will return a value that the system will use to resolve the promise.
1058 future<R> continuation_promise;
1059
1060 // lock the continuation and set it:
1061 {
1062 std::lock_guard<std::mutex> lock(_shared->_continuation_mutex);
1063
1064 if (_shared->_continuation)
1065 {
1066 return {}; // only one continuation allowed
1067 }
1068
1069 // take a weak ptr to this future's shared data. If this future goes away we'll still
1070 // have access to its result.
1071 std::weak_ptr<shared_t> weak_shared = _shared;
1072
1073 context copy_of_con = con;
1074
1075 _shared->_continuation = [func, copy_of_con, weak_shared, continuation_promise]() mutable
1076 {
1077 auto shared = weak_shared.lock();
1078
1079 // verify the parent's result is actually available (simulate available())
1080 if (shared && shared->_ev.isSet())
1081 {
1082 // copy it and dispatch it as the input to a new job:
1083 T copy_of_value = shared->_obj;
1084
1085 // Once this wrapper gets created, note that we now have 2 refereces to the continuation_promise.
1086 // To prevent this from hampering cancelation, the continuation fuction is set to nullptr
1087 // immediately after being called.
1088 auto wrapper = [func, copy_of_value, continuation_promise]() mutable
1089 {
1090 continuation_promise.resolve(func(copy_of_value, continuation_promise));
1091 };
1092
1093 jobs::dispatch(wrapper, copy_of_con);
1094 }
1095 };
1096 }
1097
1098 // maybe the future is already available?
1099 if (available())
1100 {

1110 {

1112 // Note, the user function "func" is responsible for resolving this promise.
1113 future<R> continuation_promise;
1114
1115 // lock the continuation and set it:
1116 {
1117 std::lock_guard<std::mutex> lock(_shared->_continuation_mutex);
1118
1119 if (_shared->_continuation)
1120 {
1121 return {}; // only one continuation allowed
1122 }
1123
1124 // take a weak ptr to this future's shared data. If this future goes away we'll still
1125 // have access to its result.
1126 std::weak_ptr<shared_t> weak_shared = _shared;
1127
1128 // The user task is responsible for resolving the promise.
1129 // This continuation executes the user function directly instead of dispatching it
1130 // to the job pool. This is because we expect the user function to use some external
1131 // asynchronous mechanism to resolve the promise.
1132 _shared->_continuation = [func, weak_shared, continuation_promise]() mutable
1133 {
1134 auto shared = weak_shared.lock();
1135 if (shared)
1136 {
1137 func(shared->_obj, continuation_promise);
1138 }
1139 };
1140 }
1141
1142 if (available())
1143 {
1144 fire_continuation();
1145 }
1146
1147 return continuation_promise;
