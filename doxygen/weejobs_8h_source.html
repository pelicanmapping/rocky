583 void _dispatch_delegate(std::function<bool()>& delegate, const context& context)
584 {
585 if (!_done)
586 {
587 // If we have a group semaphore, acquire it BEFORE queuing the job
588 if (context.group)
589 {
590 context.group->acquire();
591 }
592
593 if (_target_concurrency > 0)
594 {
595 std::lock_guard<std::mutex> lock(_queue_mutex);
596
597 _queue.emplace_back(detail::job{ context, delegate });
598 _queue_size++;
599
600 _metrics.pending++;
601 _metrics.total++;
602 _block.notify_one();
603 }
604 else
605 {
606 // no threads? run synchronously.
607 delegate();
608
609 if (context.group)
610 {
611 context.group->release();
612 }
613 }
614 }

620 inline bool _take_job(detail::job& output, bool lock)
621 {
622 if (lock)
623 {
624 std::lock_guard<std::mutex> lock(_queue_mutex);
625 return _take_job(output, false);
626 }
627 else if (!_done && _queue_size > 0)
628 {
629 auto ptr = _queue.end();
630 float highest_priority = -FLT_MAX;
631 for (auto iter = _queue.begin(); iter != _queue.end(); ++iter)
632 {
633 float priority = iter->ctx.priority != nullptr ?
634 iter->ctx.priority() :
635 0.0f;
636
637 if (ptr == _queue.end() || priority > highest_priority)
638 {
639 ptr = iter;
640 highest_priority = priority;
641 }
642 }
643
644 if (ptr == _queue.end())
645 ptr = _queue.begin();
646
647 output = std::move(*ptr);
648 _queue.erase(ptr);
649 _queue_size--;
650 _metrics.pending--;
651 return true;
652 }
653 return false;

908 inline void jobpool::run()
909 {
910 while (!_done)
911 {
912 detail::job next;
913 bool have_next = false;
914 {
915 if (_can_steal_work && instance()._stealing_allowed)
916 {
917 {
918 std::unique_lock<std::mutex> lock(_queue_mutex);
919
920 // work-stealing enabled: wait until any queue is non-empty
921 _block.wait(lock, [this]() { return get_metrics()->total_pending() > 0 || _done; });
922
923 if (!_done && _queue_size > 0)
924 {
925 have_next = _take_job(next, false);
926 }
927 }
928
929 if (!_done && !have_next)
930 {
931 have_next = detail::steal_job(this, next);
932 }
933 }
934 else
935 {
936 std::unique_lock<std::mutex> lock(_queue_mutex);
937
938 // wait until just our local queue is non-empty
939 _block.wait(lock, [this] { return (_queue_size > 0) || _done; });
940
941 if (!_done && _queue_size > 0)
942 {
943 have_next = _take_job(next, false);
944 }
945 }
946 }
947
948 if (have_next)
949 {
950 _metrics.running++;
951
952 auto t0 = std::chrono::steady_clock::now();
953
954 bool job_executed = next._delegate();
955
956 auto duration = std::chrono::steady_clock::now() - t0;
957
958 if (job_executed == false)
959 {
960 _metrics.canceled++;
961 }
962
963 // release the group semaphore if necessary
964 if (next.ctx.group != nullptr)
965 {
966 next.ctx.group->release();
967 }
968
969 _metrics.running--;
970 }
971
972 // See if we no longer need this thread because the
973 // target concurrency has been reduced
974 std::lock_guard<std::mutex> lock(_quit_mutex);
975
976 if (_target_concurrency < _metrics.concurrency)
977 {
978 _metrics.concurrency--;
979 break;
980 }
981 }
982 }

1005 inline void jobpool::stop_threads()
1006 {
1007 _done = true;
1008
1009 // Clear out the queue
1010 std::lock_guard<std::mutex> lock(_queue_mutex);
1011
1012 // reset any group semaphores so that JobGroup.join()
1013 // will not deadlock.
1014 for (auto& queuedjob : _queue)
1015 {
1016 if (queuedjob.ctx.group != nullptr)
1017 {
1018 queuedjob.ctx.group->release();
1019 }
1020 }
1021 _queue.clear();
1022 _queue_size = 0;
1023
1024 // wake up all threads so they can exit
1025 _block.notify_all();
1026 }

1043 // steal a job from another jobpool's queue (other than "thief").
1044 inline bool detail::steal_job(jobpool* thief, detail::job& stolen)
1045 {
1046 jobpool* pool_with_most_jobs = nullptr;
1047 {
1048 std::lock_guard<std::mutex> lock(instance()._pools_mutex);
1049
1050 std::size_t max_num_jobs = 0u;
1051 for (auto pool : instance()._pools)
1052 {
1053 if (pool != thief)
1054 {
1055 if (static_cast<std::size_t>(pool->_queue_size) > max_num_jobs)
1056 {
1057 max_num_jobs = pool->_queue_size;
1058 pool_with_most_jobs = pool;
1059 }
1060 }
1061 }
1062 }
1063
1064 if (pool_with_most_jobs)
1065 {
1066 return pool_with_most_jobs->_take_job(stolen, true);
1067 }
1068
1069 return false;

1072 template<typename T>
1073 template<typename F, typename R>
1074 inline future<R> future<T>::then_dispatch(F func, const context& con)
1075 {
1076 // The future result of F.
1077 // In this case, the continuation task will return a value that the system will use to resolve the promise.
1078 future<R> continuation_promise;
1079
1080 // lock the continuation and set it:
1081 {
1082 std::lock_guard<std::mutex> lock(_shared->_continuation_mutex);
1083
1084 if (_shared->_continuation)
1085 {
1086 return {}; // only one continuation allowed
1087 }
1088
1089 // take a weak ptr to this future's shared data. If this future goes away we'll still
1090 // have access to its result.
1091 std::weak_ptr<shared_t> weak_shared = _shared;
1092
1093 context copy_of_con = con;
1094
1095 _shared->_continuation = [func, copy_of_con, weak_shared, continuation_promise]() mutable
1096 {
1097 auto shared = weak_shared.lock();
1098
1099 // verify the parent's result is actually available (simulate available())
1100 if (shared && shared->_ev.isSet())
1101 {
1102 // copy it and dispatch it as the input to a new job:
1103 T copy_of_value = shared->_obj;
1104
1105 // Once this wrapper gets created, note that we now have 2 refereces to the continuation_promise.
1106 // To prevent this from hampering cancelation, the continuation fuction is set to nullptr
1107 // immediately after being called.
1108 auto wrapper = [func, copy_of_value, continuation_promise]() mutable

1130 {

1132 // Note, the user function "func" is responsible for resolving this promise.
1133 future<R> continuation_promise;
1134
1135 // lock the continuation and set it:
1136 {
1137 std::lock_guard<std::mutex> lock(_shared->_continuation_mutex);
1138
1139 if (_shared->_continuation)
1140 {
1141 return {}; // only one continuation allowed
1142 }
1143

1145 // have access to its result.
1146 std::weak_ptr<shared_t> weak_shared = _shared;
1147
1148 // The user task is responsible for resolving the promise.
1149 // This continuation executes the user function directly instead of dispatching it
1150 // to the job pool. This is because we expect the user function to use some external
1151 // asynchronous mechanism to resolve the promise.
1152 _shared->_continuation = [func, weak_shared, continuation_promise]() mutable
1153 {
1154 auto shared = weak_shared.lock();

1156 {
1157 func(shared->_obj, continuation_promise);
1158 }
1159 };
1160 }
1161
1162 if (available())
1163 {
1164 fire_continuation();
1165 }
1166
1167 return continuation_promise;
