584  {

620  inline bool _take_job(detail::job& output, bool lock)
621  {
622  if (lock)
623  {
624  std::lock_guard<std::mutex> lock(_queue_mutex);
625  return _take_job(output, false);
626  }
627  else if (!_done && _queue_size > 0)
628  {
629  auto ptr = _queue.end();
630  float highest_priority = -FLT_MAX;
631  for (auto iter = _queue.begin(); iter != _queue.end(); ++iter)
632  {
633  float priority = iter->ctx.priority != nullptr ?
634  iter->ctx.priority() :
635  0.0f;
636 
637  if (ptr == _queue.end() || priority > highest_priority)
638  {
639  ptr = iter;
640  highest_priority = priority;
641  }
642  }
643 
644  if (ptr == _queue.end())
645  ptr = _queue.begin();
646 
647  output = std::move(*ptr);
648  _queue.erase(ptr);
649  _queue_size--;
650  _metrics.pending--;
651  return true;
652  }
653  return false;

908  inline void jobpool::run()
909  {
910  while (!_done)
911  {
912  detail::job next;
913  bool have_next = false;
914  {
915  if (_can_steal_work && instance()._stealing_allowed)
916  {
917  {
918  std::unique_lock<std::mutex> lock(_queue_mutex);
919 
920  // work-stealing enabled: wait until any queue is non-empty
921  _block.wait(lock, [this]() { return get_metrics()->total_pending() > 0 || _done; });
922 
923  if (!_done && _queue_size > 0)
924  {
925  have_next = _take_job(next, false);
926  }
927  }
928 
929  if (!_done && !have_next)
930  {
931  have_next = detail::steal_job(this, next);
932  }
933  }
934  else
935  {
936  std::unique_lock<std::mutex> lock(_queue_mutex);
937 
938  // wait until just our local queue is non-empty
939  _block.wait(lock, [this] { return (_queue_size > 0) || _done; });
940 
941  if (!_done && _queue_size > 0)
942  {
943  have_next = _take_job(next, false);
944  }
945  }
946  }
947 
948  if (have_next)
949  {
950  _metrics.running++;
951 
952  auto t0 = std::chrono::steady_clock::now();
953 
954  bool job_executed = next._delegate();
955 
956  auto duration = std::chrono::steady_clock::now() - t0;
957 
958  if (job_executed == false)
959  {
960  _metrics.canceled++;
961  }
962 
963  // release the group semaphore if necessary
964  if (next.ctx.group != nullptr)
965  {
966  next.ctx.group->release();
967  }
968 
969  _metrics.running--;
970  }
971 
972  // See if we no longer need this thread because the
973  // target concurrency has been reduced
974  std::lock_guard<std::mutex> lock(_quit_mutex);
975 
976  if (_target_concurrency < _metrics.concurrency)
977  {
978  _metrics.concurrency--;
979  break;
980  }
981  }
982  }
