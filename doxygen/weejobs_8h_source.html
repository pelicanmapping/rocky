581 void _dispatch_delegate(std::function<bool()>& delegate, const context& context)
582 {
583 if (!_done)
584 {
585 // If we have a group semaphore, acquire it BEFORE queuing the job
586 if (context.group)
587 {
588 context.group->acquire();
589 }
590
591 if (_target_concurrency > 0)
592 {
593 std::lock_guard<std::mutex> lock(_queue_mutex);
594
595 _queue.emplace_back(detail::job{ context, delegate });
596
597 _metrics.pending++;
598 _metrics.total++;
599 _block.notify_one();
600 }
601 else
602 {
603 // no threads? run synchronously.
604 delegate();
605
606 if (context.group)
607 {
608 context.group->release();
609 }
610 }

617 inline bool _take_job(detail::job& output, bool lock)
618 {
619 if (lock)
620 {
621 std::lock_guard<std::mutex> lock(_queue_mutex);
622 return _take_job(output, false);
623 }
624 else if (!_done && !_queue.empty())
625 {
626 auto ptr = _queue.end();
627 float highest_priority = -FLT_MAX;
628 for (auto iter = _queue.begin(); iter != _queue.end(); ++iter)
629 {
630 float priority = iter->ctx.priority != nullptr ?
631 iter->ctx.priority() :
632 0.0f;
633
634 if (ptr == _queue.end() || priority > highest_priority)
635 {
636 ptr = iter;
637 highest_priority = priority;
638 }
639 }
640
641 if (ptr == _queue.end())
642 ptr = _queue.begin();
643
644 output = std::move(*ptr);
645 if (_queue.size() > 1)
646 *ptr = std::move(_queue.back());
647 _queue.resize(_queue.size() - 1);
648
649 _metrics.pending--;
650 return true;
651 }
652 return false;
653 }

907 inline void jobpool::run()
908 {
909 while (!_done)
910 {
911 detail::job next;
912 bool have_next = false;
913 {
914 if (_can_steal_work && instance()._stealing_allowed)
915 {
916 {
917 std::unique_lock<std::mutex> lock(_queue_mutex);
918
919 // work-stealing enabled: wait until any queue is non-empty
920 _block.wait(lock, [this]() { return get_metrics()->total_pending() > 0 || _done; });
921
922 if (!_done && !_queue.empty())
923 {
924 have_next = _take_job(next, false);
925 }
926 }
927
928 if (!_done && !have_next)
929 {
930 have_next = detail::steal_job(this, next);
931 }
932 }
933 else
934 {
935 std::unique_lock<std::mutex> lock(_queue_mutex);
936
937 // wait until just our local queue is non-empty
938 _block.wait(lock, [this] { return !_queue.empty() || _done; });
939
940 if (!_done && !_queue.empty())
941 {
942 have_next = _take_job(next, false);
943 }
944 }
945 }
946
947 if (have_next)
948 {
949 _metrics.running++;
950
951 auto t0 = std::chrono::steady_clock::now();
952
953 bool job_executed = next._delegate();
954
955 auto duration = std::chrono::steady_clock::now() - t0;
956
957 if (job_executed == false)
958 {
959 _metrics.canceled++;
960 }
961
962 // release the group semaphore if necessary
963 if (next.ctx.group != nullptr)
964 {
965 next.ctx.group->release();
966 }
967
968 _metrics.running--;
969 }
970
971 // See if we no longer need this thread because the
972 // target concurrency has been reduced
973 std::lock_guard<std::mutex> lock(_quit_mutex);
974
975 if (_target_concurrency < _metrics.concurrency)
976 {
977 _metrics.concurrency--;
978 break;
979 }
980 }
981 }

1004 inline void jobpool::stop_threads()
1005 {
1006 _done = true;
1007
1008 // Clear out the queue
1009 std::lock_guard<std::mutex> lock(_queue_mutex);
1010
1011 // reset any group semaphores so that JobGroup.join()
1012 // will not deadlock.
1013 for (auto& queuedjob : _queue)
1014 {
1015 if (queuedjob.ctx.group != nullptr)
1016 {
1017 queuedjob.ctx.group->release();
1018 }
1019 }
1020 _queue.clear();
1021
1022 // wake up all threads so they can exit
1023 _block.notify_all();
1024 }

1070 template<typename T>
1071 template<typename F, typename R>
1072 inline future<R> future<T>::then_dispatch(F func, const context& con)
1073 {
1074 // The future result of F.
1075 // In this case, the continuation task will return a value that the system will use to resolve the promise.
1076 future<R> continuation_promise;
1077
1078 // lock the continuation and set it:
1079 {
1080 std::lock_guard<std::mutex> lock(_shared->_continuation_mutex);
1081
1082 if (_shared->_continuation)
1083 {
1084 return {}; // only one continuation allowed
1085 }
1086
1087 // take a weak ptr to this future's shared data. If this future goes away we'll still
1088 // have access to its result.
1089 std::weak_ptr<shared_t> weak_shared = _shared;
1090
1091 context copy_of_con = con;
1092
1093 _shared->_continuation = [func, copy_of_con, weak_shared, continuation_promise]() mutable
1094 {
1095 auto shared = weak_shared.lock();
1096
1097 // verify the parent's result is actually available (simulate available())
1098 if (shared && shared->_ev.isSet())
1099 {
1100 // copy it and dispatch it as the input to a new job:
1101 T copy_of_value = shared->_obj;
1102
1103 // Once this wrapper gets created, note that we now have 2 refereces to the continuation_promise.
1104 // To prevent this from hampering cancelation, the continuation fuction is set to nullptr
1105 // immediately after being called.
1106 auto wrapper = [func, copy_of_value, continuation_promise]() mutable
1107 {
1108 continuation_promise.resolve(func(copy_of_value, continuation_promise));

1134 {

1136
1137 if (_shared->_continuation)
1138 {

1149 // asynchronous mechanism to resolve the promise.

1157 };
1158 }

1162 fire_continuation();
1163 }
