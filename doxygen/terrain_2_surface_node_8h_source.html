74 inline bool SurfaceNode::isVisible(vsg::RecordTraversal& rv) const
75 {
76 auto* state = rv.getState();
77
78 // bounding box visibility check; this is much tighter than the bounding
79 // sphere. _frustumStack.top() contains the frustum in world coordinates.
80 // https://github.com/vsg-dev/VulkanSceneGraph/blob/master/include/vsg/vk/State.h#L267
81 // The first 8 points in _worldPoints are the 8 corners of the bounding box
82 // in world coordinates.
83 // Note: POLYTOPE_SIZE is defined in vsg plane.h
84 auto& frustum = state->_frustumStack.top();
85 int p;
86 for (int f = 0; f < POLYTOPE_SIZE; ++f)
87 {
88 for (p = 0; p < 8; ++p)
89 if (vsg::distance(frustum.face[f], _worldPoints[p]) > 0.0) // visible?
90 break;
91
92 if (p == 8)
93 return false;
94 }
95
96 // Horizon culling:
97 auto& viewlocal = _viewlocal[state->_commandBuffer->viewID];
98 if (!viewlocal.horizon)
99 {
100 rv.getValue("rocky.horizon", viewlocal.horizon);
101 }
102 if (viewlocal.horizon)
103 {
104 if (_horizonCullingPoint_valid)
105 {
106 return viewlocal.horizon->isVisible(_horizonCullingPoint);
107 }
108 else
109 {
110 for (int p = 0; p < 4; ++p)
111 {
112 auto& wp = _worldPoints[p];
113 if (viewlocal.horizon->isVisible(wp.x, wp.y, wp.z))
114 return true;
115 }
116 return false;
117 }
118 }
119
120 return true;
121 }
122}
