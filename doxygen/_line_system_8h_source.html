95 void LineGeometry::set(const std::vector<VEC3_T>& verts, Line::Topology topology, std::size_t staticStorage)
96 {
97 const vsg::vec4 defaultColor = { 1.0f, 1.0f, 1.0f, 1.0f };
98
99 if (!_current)
100 {
101 std::size_t verts_to_allocate = (staticStorage > 0 ? staticStorage : verts.size());
102 std::size_t indices_to_allocate =
103 topology == Line::Topology::Strip ? (verts_to_allocate - 1) * 6 :
104 (verts_to_allocate / 2) * 6; // Segments
105
106 _current = vsg::vec3Array::create(verts_to_allocate * 4);
107 _current->properties.dataVariance = vsg::DYNAMIC_DATA;
108
109 _previous = vsg::vec3Array::create(verts_to_allocate * 4);
110 _previous->properties.dataVariance = vsg::DYNAMIC_DATA;
111
112 _next = vsg::vec3Array::create(verts_to_allocate * 4);
113 _next->properties.dataVariance = vsg::DYNAMIC_DATA;
114
115 _colors = vsg::vec4Array::create(verts_to_allocate * 4);
116 _colors->properties.dataVariance = vsg::DYNAMIC_DATA;
117
118 assignArrays({ _current, _previous, _next, _colors });
119
120 _indices = vsg::uintArray::create((verts_to_allocate - 1) * 6);
121 _indices->properties.dataVariance = vsg::DYNAMIC_DATA;
122 assignIndices(_indices);
123 }
124
125 auto* current = (_current->data());
126 auto* prev = (_previous->data());
127 auto* next = (_next->data());
128 auto* color = (_colors->data());
129 auto* indicies = (_indices->data());
130 int i_ptr = 0;
131
132 if (topology == Line::Topology::Strip)
133 {
134 for (unsigned i = 0; i < verts.size(); ++i)
135 {
136 bool first = i == 0;
137 bool last = (i == verts.size() - 1);
138
139 for (int n = 0; n < 4; ++n)
140 {
141 (_previous->data())[i * 4 + n] = first ? verts[i] : verts[i - 1];
142 (_next->data())[i * 4 + n] = last ? verts[i] : verts[i + 1];
143 (_current->data())[i * 4 + n] = verts[i];
144 (_colors->data())[i * 4 + n] = defaultColor;
145 }
146
147 if (!first)
148 {
149 auto e = (i - 1) * 4 + 2;
150 (_indices->data())[i_ptr++] = e + 3;
151 (_indices->data())[i_ptr++] = e + 1;
152 (_indices->data())[i_ptr++] = e + 0; // provoking vertex
153 (_indices->data())[i_ptr++] = e + 2;
154 (_indices->data())[i_ptr++] = e + 3;
155 (_indices->data())[i_ptr++] = e + 0; // provoking vertex
156 }
157 }
158 }
159 else // topology == Topology::Segments
160 {
161 ROCKY_SOFT_ASSERT_AND_RETURN((verts.size() & 0x1) == 0, void(), "Lines with 'Segment' topology must have an even number of vertices");
162
163 for (unsigned i = 0; i < verts.size(); ++i)
164 {
165 bool even = (i & 0x1) == 0;
166
167 for (int n = 0; n < 4; ++n)
168 {
169 if (even) // beginning of segment
170 {
171 prev[i * 4 + n] = verts[i];
172 next[i * 4 + n] = verts[i + 1];
173 }
174 else // end of segment
175 {
176 prev[i * 4 + n] = verts[i - 1];
177 next[i * 4 + n] = verts[i];
178 }
179
180 current[i * 4 + n] = verts[i];
181 color[i * 4 + n] = defaultColor;
182 }
183
184 if (even)
185 {
186 auto e = (i) * 4 + 2;
187 indicies[i_ptr++] = e + 3;
188 indicies[i_ptr++] = e + 1;
189 indicies[i_ptr++] = e + 0; // provoking vertex
190 indicies[i_ptr++] = e + 2;
191 indicies[i_ptr++] = e + 3;
192 indicies[i_ptr++] = e + 0; // provoking vertex
193 }
194 }
195 }
196
197 _drawCommand->firstIndex = 0;
198 _drawCommand->indexCount = i_ptr;
199
200 _current->dirty();
201 _previous->dirty();
202 _next->dirty();
203 _colors->dirty();
204 _indices->dirty();
205 }
