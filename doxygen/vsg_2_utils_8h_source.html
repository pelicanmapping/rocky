98 template<typename vec_type>
99 inline void expandBy(vsg::dsphere& bs, const vec_type& v)
100 {
101 if (bs.valid())
102 {
103 auto dv = vsg::dvec3(v) - bs.center;
104 auto r = vsg::length(dv);
105 if (r > bs.radius)
106 {
107 auto dr = (r - bs.radius) * 0.5;
108 bs.center += dv * (dr / r);
109 bs.radius += dr;
110 }
111 }
112 else
113 {
114 bs.center = v;
115 bs.radius = 0.0;
116 }
117 }

313 // Convert a vsg::Data structure to an Image if possible
314 inline Result<std::shared_ptr<Image>> makeImageFromVSG(vsg::ref_ptr<vsg::Data> data)
315 {
316 if (!data)
317 return Status(Status::ResourceUnavailable, "Data is empty");
318
319 // TODO: move this into a utility somewhere
320 auto vkformat = data->properties.format;
321
322 Image::PixelFormat format =
323 vkformat == VK_FORMAT_R8_UNORM ? Image::R8_UNORM :
324 vkformat == VK_FORMAT_R8_SRGB ? Image::R8_SRGB :
325 vkformat == VK_FORMAT_R8G8_UNORM ? Image::R8G8_UNORM :
326 vkformat == VK_FORMAT_R8G8_SRGB ? Image::R8G8_SRGB :
327 vkformat == VK_FORMAT_R8G8B8_UNORM ? Image::R8G8B8_UNORM :
328 vkformat == VK_FORMAT_R8G8B8_SRGB ? Image::R8G8B8_SRGB :
329 vkformat == VK_FORMAT_R8G8B8A8_UNORM ? Image::R8G8B8A8_UNORM :
330 vkformat == VK_FORMAT_R8G8B8A8_SRGB ? Image::R8G8B8A8_SRGB :
331 vkformat == VK_FORMAT_R16_UNORM ? Image::R16_UNORM :
332 vkformat == VK_FORMAT_R32_SFLOAT ? Image::R32_SFLOAT :
333 vkformat == VK_FORMAT_R64_SFLOAT ? Image::R64_SFLOAT :
334 Image::UNDEFINED;
335
336 if (format == Image::UNDEFINED)
337 {
338 return Status(Status::ResourceUnavailable, "Unsupported image format");
339 }
340
341 auto image = Image::create(
342 format,
343 data->width(),
344 data->height(),
345 data->depth());
346
347 memcpy(image->data<uint8_t>(), data->dataPointer(), image->sizeInBytes());
348
349 if (data->properties.origin == vsg::TOP_LEFT)
350 {
351 image->flipVerticalInPlace();
352 }
353
354 return Result(image);
355 }
