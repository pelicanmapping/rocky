54 template<typename E>
55 inline void propagate(E& e, bool forceRefresh = false)
56 {
57 // only process events for the window we are interested in, and if the event wasn't handled
58 // (say, by another wrapper connected to another view)
59 if (!e.handled && ((_window == nullptr) || (e.window.ref_ptr() == _window)))
60 {
61 // active the context associated with this window/view
62 if (_imguiContext)
63 {
64 ImGui::SetCurrentContext(_imguiContext);
65 }
66
67 vsgImGui::SendEventsToImGui::apply(e);
68
69 if (_vsgContext && (e.handled || forceRefresh))
70 {
71 _vsgContext->requestFrame();
72 }
73 }
74 }

76 void apply(vsg::ButtonPressEvent& e) override { propagate(e); }
77 void apply(vsg::ButtonReleaseEvent& e) override { propagate(e); }
