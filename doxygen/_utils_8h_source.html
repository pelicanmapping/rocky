53 std::string format( const std::string& format, Args... args) {
54 int size_s = std::snprintf( nullptr, 0, format.c_str(), args... ) + 1; // Extra space for '\0'

295 public:

297
298 ring_buffer(int size = 8, bool overwrite_when_full_ = false) :
299 _size(size), _buffer(size), overwrite_when_full(overwrite_when_full_){}
300
301 bool overwrite_when_full = false;
302
305 void resize(std::size_t newSize) {
306 _buffer.clear();
307 _buffer.resize(newSize);
308 _size = newSize;
309 _readIndex = { 0 };
310 _writeIndex = { 0 };
311 }
312
313 bool push(const T& obj) {
314 bool is_full = full();
315 if (is_full && !overwrite_when_full) return false;
316 _buffer[_writeIndex] = obj;
317 if (!is_full) _writeIndex.exchange((_writeIndex + 1) % _size);
318 notify();
319 return true;
320 }
321
322 bool pop(T& obj) {
323 if (_readIndex == _writeIndex) return false;
324 obj = std::move(_buffer[_readIndex]);
325 _readIndex.exchange((_readIndex + 1) % _size);
326 return true;
327 }
328
329 T& peek() {
330 return _buffer[_readIndex];
331 }
332
333 const T& peek() const {
334 return _buffer[_readIndex];
335 }
336
337 protected:
338 virtual void notify() { }
339
340 bool empty() const {
341 return _readIndex == _writeIndex;
342 }
343
344 bool full() const {
345 return (_writeIndex + 1) % _size == _readIndex;
346 }
347
348 private:
349 std::atomic_int _readIndex = { 0 };
350 std::atomic_int _writeIndex = { 0 };
351 int _size;
352 std::vector<T> _buffer;
353 };

482 class BackgroundServices
483 {
484 public:
485 using Function = std::function<void(jobs::cancelable&)>;
486
487 std::mutex mutex;
488 std::vector<jobs::future<bool>> futures;
489 jobs::detail::semaphore semaphore;
490
491 void start(const std::string& name, Function function)
492 {
493 ROCKY_SOFT_ASSERT_AND_RETURN(function, void());
494 ROCKY_SOFT_ASSERT_AND_RETURN(!name.empty(), void());
495
496 std::lock_guard lock(mutex);
497
498 // wrap with a delegate function so we can control the semaphore
499 auto delegate = [this, function, name](jobs::cancelable& cancelable) -> bool
500 {
501 ++semaphore;
502 function(cancelable);
503 --semaphore;
504 return true;
505 };
506
507 jobs::context context{ name, jobs::get_pool(name, 1) };
508 futures.emplace_back(jobs::dispatch(delegate, context));
509 }
510
511 void quit()
512 {
513 std::lock_guard lock(mutex);
514
515 // tell all tasks to cancel
516 for (auto& f : futures)
517 f.abandon();
518
519 // block until all the background tasks exit.
520 semaphore.join();
521
522 futures.clear();
523 }
524 };
