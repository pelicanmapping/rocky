49 template<typename ... Args>
50 std::string format( const std::string& format, Args... args) {
51 int size_s = std::snprintf( nullptr, 0, format.c_str(), args... ) + 1; // Extra space for '\0'
52 if( size_s <= 0 ) { throw std::runtime_error( "Error during formatting." ); }
53 auto size = static_cast<size_t>( size_s );
54 std::unique_ptr<char[]> buf( new char[ size ] );
55 std::snprintf( buf.get(), size, format.c_str(), args ... );
56 return std::string( buf.get(), buf.get() + size - 1 ); // We don't want the '\0' inside
57 }
