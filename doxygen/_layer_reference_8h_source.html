129 void get(const Config& conf, const std::string& tag)
130 {
131 // first try to store the name of another layer:
132 conf.get(tag, _externalLayerName);
133
134 if (!_externalLayerName.has_value())
135 {
136 // next try to find a child called (tag) and try to make the layer
137 // from it's children:
138 if (conf.hasChild(tag) && conf.child(tag).children().size() >= 1)
139 {
140 const Config& tag_content = *conf.child(tag).children().begin();
141 {
142 osg::ref_ptr<Layer> layer = Layer::create(tag_content);
143 if (layer && std::dynamic_pointer_cast<T>(layer))
144 {
145 _embeddedOptions = TypedOptions(tag_content);
146 }
147 }
148 }
149
150 // failing that, try each child of the config.
151 if (!_embeddedOptions.has_value())
152 {
153 for(auto& conf : conf.children())
154 {
155 std::shared_ptr<Layer> layer = Layer::create(conf);
156 if (layer && std::dynamic_pointer_cast<T>(layer))
157 {
158 _embeddedOptions = TypedOptions(conf);
159 break;
160 }
161 }
162 }
