133  void Geometry::iterator<T>::fetch() {
134  _next = nullptr;
135  if (_queue.empty())
136  return;
137  T* current = _queue.front();
138  _queue.pop();
139  bool is_multi =
140  current->type == Geometry::Type::MultiLineString ||
141  current->type == Geometry::Type::MultiPoints ||
142  current->type == Geometry::Type::MultiPolygon;
143 
144  if (is_multi && _traverse_multi)
145  {
146  for (auto& part : current->parts)
147  _queue.push(&part);
148  fetch();
149  }
150  else
151  {
152  _next = current;
153  if (current->type == Type::Polygon && _traverse_polygon_holes)
154  {
155  for (auto& ring : current->parts)
