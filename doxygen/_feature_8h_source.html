131 template<class T>
132 void Geometry::iterator<T>::fetch() {
133 _next = nullptr;
134 if (_queue.empty())
135 return;
136 T* current = _queue.front();
137 _queue.pop();
138 bool is_multi =
139 current->type == Geometry::Type::MultiLineString ||
140 current->type == Geometry::Type::MultiPoints ||
141 current->type == Geometry::Type::MultiPolygon;
142
143 if (is_multi && _traverse_multi)
144 {
145 for (auto& part : current->parts)
146 _queue.push(&part);
147 fetch();
148 }
149 else
150 {
151 _next = current;
152 if (current->type == Type::Polygon && _traverse_polygon_holes)
153 {
154 for (auto& ring : current->parts)
155 _queue.push(&ring);
156 }
157 }
