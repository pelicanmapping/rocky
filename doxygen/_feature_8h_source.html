126 template<class T>
127 void Geometry::iterator<T>::fetch() {
128 _next = nullptr;
129 if (_queue.empty())
130 return;
131 T* current = _queue.front();
132 _queue.pop();
133 bool is_multi =
134 current->type == Geometry::Type::MultiLineString ||
135 current->type == Geometry::Type::MultiPoints ||
136 current->type == Geometry::Type::MultiPolygon;
137
138 if (is_multi && _traverse_multi)
139 {
140 for (auto& part : current->parts)
141 _queue.push(&part);
142 fetch();
143 }
144 else
145 {
146 _next = current;
147 if (current->type == Type::Polygon && _traverse_polygon_holes)
148 {
149 for (auto& ring : current->parts)
150 _queue.push(&ring);
151 }
152 }
