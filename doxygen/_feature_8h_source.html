147 template<class T>
148 void Geometry::iterator<T>::fetch() {
149 _next = nullptr;
150 if (_queue.empty())
151 return;
152 T* current = _queue.front();
153 _queue.pop();
154 bool is_multi =
155 current->type == Geometry::Type::MultiLineString ||
156 current->type == Geometry::Type::MultiPoints ||
157 current->type == Geometry::Type::MultiPolygon;
158
159 if (is_multi && _traverse_multi)
160 {
161 for (auto& part : current->parts)
162 _queue.push(&part);
163 fetch();
164 }
165 else
166 {
167 _next = current;
168 if (current->type == Type::Polygon && _traverse_polygon_holes)
169 {
170 for (auto& ring : current->parts)
171 _queue.push(&ring);
172 }
173 }
174 }
