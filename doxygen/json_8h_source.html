98 inline bool get_to(const json& obj, const char* name, T& var)
99 {
100 if (name != nullptr && obj.is_object() && !obj.is_null())
101 {
102 JSON_TRY
103 {
104 auto i = obj.find(name);
105 if (i != obj.end())

107 i->get_to(var);
108 return true;
109 }
110 }
111 JSON_CATCH{ }
112 }
