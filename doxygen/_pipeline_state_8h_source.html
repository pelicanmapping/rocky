86 class DescriptorBufferEx : public vsg::Inherit<vsg::DescriptorBuffer, DescriptorBufferEx>
87 {
88 public:
89 explicit DescriptorBufferEx(const vsg::BufferInfoList& in_bufferInfoList, uint32_t dstBinding, uint32_t dstArrayElement, VkDescriptorType descriptorType, VkBufferUsageFlags in_additionalUsageFlags) :
90 Inherit(in_bufferInfoList, dstBinding, dstArrayElement, descriptorType),
91 additionalUsageFlags(in_additionalUsageFlags)
92 {
93 //nop
94 }
95
96 VkBufferUsageFlags additionalUsageFlags = 0;
97
98
99 void compile(vsg::Context& context) override
100 {
101 if (bufferInfoList.empty()) return;
102
103 auto transferTask = context.transferTask.get();
104
105 VkBufferUsageFlags bufferUsageFlags = additionalUsageFlags;
106 switch (descriptorType)
107 {
108 case VK_DESCRIPTOR_TYPE_UNIFORM_BUFFER:
109 case VK_DESCRIPTOR_TYPE_UNIFORM_BUFFER_DYNAMIC:
110 bufferUsageFlags |= VK_BUFFER_USAGE_UNIFORM_BUFFER_BIT;
111 break;
112 case VK_DESCRIPTOR_TYPE_STORAGE_BUFFER:
113 case VK_DESCRIPTOR_TYPE_STORAGE_BUFFER_DYNAMIC:
114 bufferUsageFlags |= VK_BUFFER_USAGE_STORAGE_BUFFER_BIT;
115 break;
116 default:
117 break;
118 }
119
120 bool requiresAssignmentOfBuffers = false;
121 for (auto& bufferInfo : bufferInfoList)
122 {
123 if (bufferInfo->buffer == nullptr) requiresAssignmentOfBuffers = true;
124 }
125
126 auto deviceID = context.deviceID;
127
128 if (requiresAssignmentOfBuffers)
129 {
130 VkDeviceSize alignment = 4;
131 const auto& limits = context.device->getPhysicalDevice()->getProperties().limits;
132 if (bufferUsageFlags & VK_BUFFER_USAGE_UNIFORM_BUFFER_BIT)
133 alignment = limits.minUniformBufferOffsetAlignment;
134 else if (bufferUsageFlags & VK_BUFFER_USAGE_STORAGE_BUFFER_BIT)
135 alignment = limits.minStorageBufferOffsetAlignment;
136
137 VkDeviceSize totalSize = 0;
138
139 // compute the total size of BufferInfo that needs to be allocated.
140 {
141 VkDeviceSize offset = 0;
142 for (auto& bufferInfo : bufferInfoList)
143 {
144 if (bufferInfo->data && !bufferInfo->buffer)
145 {
146 totalSize = offset + bufferInfo->data->dataSize();
147 offset = (alignment == 1 || (totalSize % alignment) == 0) ? totalSize : ((totalSize / alignment) + 1) * alignment;
148 if (bufferInfo->data->dynamic() || transferTask) bufferUsageFlags |= VK_BUFFER_USAGE_TRANSFER_DST_BIT;
149 }
150 }
151 }
152
153 // if required allocate the buffer and reserve slots in it for the BufferInfo
154 if (totalSize > 0)
155 {
156 auto buffer = vsg::Buffer::create(totalSize, bufferUsageFlags, VK_SHARING_MODE_EXCLUSIVE);
157 for (auto& bufferInfo : bufferInfoList)
158 {
159 if (bufferInfo->data && !bufferInfo->buffer)
160 {
161 auto [allocated, offset] = buffer->reserve(bufferInfo->data->dataSize(), alignment);
162 if (allocated)
163 {
164 bufferInfo->buffer = buffer;
165 bufferInfo->offset = offset;
166 bufferInfo->range = bufferInfo->data->dataSize();
167 }
168 else
169 {
170 vsg::warn("DescriptorBuffer::compile(..) unable to allocate bufferInfo within associated Buffer.");
171 }
172 }
173 }
174 }
175 }
176
177 for (auto& bufferInfo : bufferInfoList)
178 {
179 if (bufferInfo->buffer)
180 {
181 if (bufferInfo->buffer->compile(context.device))
182 {
183 if (bufferInfo->buffer->getDeviceMemory(deviceID) == nullptr)
184 {
185 auto memRequirements = bufferInfo->buffer->getMemoryRequirements(deviceID);
186 VkMemoryPropertyFlags flags = VK_MEMORY_PROPERTY_HOST_VISIBLE_BIT | VK_MEMORY_PROPERTY_HOST_COHERENT_BIT;
187 auto [deviceMemory, offset] = context.deviceMemoryBufferPools->reserveMemory(memRequirements, flags);
188 if (deviceMemory)
189 {
190 bufferInfo->buffer->bind(deviceMemory, offset);
191 }
192 else
193 {
194 throw vsg::Exception{ "Error: DescriptorBuffer::compile(..) failed to allocate buffer from deviceMemoryBufferPools.", VK_ERROR_OUT_OF_DEVICE_MEMORY };
195 }
196 }
197 }
198
199 if (!transferTask && bufferInfo->data && bufferInfo->data->getModifiedCount(bufferInfo->copiedModifiedCounts[deviceID]))
200 {
201 bufferInfo->copyDataToBuffer(context.deviceID);
202 }
203 }
204 }
205
206 if (transferTask) transferTask->assign(bufferInfoList);
207 }
208 };
