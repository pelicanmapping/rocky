22 static void addViewDependentData(vsg::ref_ptr<vsg::ShaderSet> shaderSet, VkShaderStageFlags stageFlags)
23 {
24 // override it because we're getting weird VK errors -gw
25 stageFlags = VK_SHADER_STAGE_VERTEX_BIT | VK_SHADER_STAGE_FRAGMENT_BIT;
26
27 // VSG view-dependent data. You must include it all even if you only intend to use
28 // one of the uniforms.
29 shaderSet->customDescriptorSetBindings.push_back(
30 vsg::ViewDependentStateBinding::create(VSG_VIEW_DEPENDENT_DESCRIPTOR_SET_INDEX));
31
32 shaderSet->addDescriptorBinding(
33 "vsg_lights", "",
34 VSG_VIEW_DEPENDENT_DESCRIPTOR_SET_INDEX,
35 VSG_VIEW_DEPENDENT_LIGHTS_BINDING,
36 VK_DESCRIPTOR_TYPE_UNIFORM_BUFFER, 1,
37 stageFlags, {});
38
39 // VSG viewport state
40 shaderSet->addDescriptorBinding(
41 "vsg_viewports", "",
42 VSG_VIEW_DEPENDENT_DESCRIPTOR_SET_INDEX,
43 VSG_VIEW_DEPENDENT_VIEWPORTS_BINDING,
44 VK_DESCRIPTOR_TYPE_STORAGE_BUFFER, 1,
45 stageFlags, {});
46 }
