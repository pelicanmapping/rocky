238 void Image::read_bilinear(Pixel& pixel, float u, float v, unsigned layer) const
239 {
240 u = clamp(u, 0.0f, 1.0f);
241 v = clamp(v, 0.0f, 1.0f);
242
243 float sizeS = (float)(width() - 1);
244 float s = u * sizeS;
245 float s0 = std::max(std::floor(s), 0.0f);
246 float s1 = std::min(s0 + 1.0f, sizeS);
247 float smix = s0 < s1 ? (s - s0) / (s1 - s0) : 0.0f;
248
249 float sizeT = (float)(height() - 1);
250 float t = v * sizeT;
251 float t0 = std::max(std::floor(t), 0.0f);
252 float t1 = std::min(t0 + 1.0f, sizeT);
253 float tmix = t0 < t1 ? (t - t0) / (t1 - t0) : 0.0f;
254
255 Pixel UL, UR, LL, LR;
256 read(UL, (unsigned)s0, (unsigned)t0, layer);
257 read(UR, (unsigned)s1, (unsigned)t0, layer);
258 read(LL, (unsigned)s0, (unsigned)t1, layer);
259 read(LR, (unsigned)s1, (unsigned)t1, layer);
260
261 Pixel TOP = UL * (1.0f - smix) + UR * smix;
262 Pixel BOT = LL * (1.0f - smix) + LR * smix;
263 pixel = TOP * (1.0f - tmix) + BOT * tmix;
264 }
