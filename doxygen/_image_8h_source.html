253 void Image::read_bilinear(Pixel& pixel, float u, float v, unsigned r) const
254 {
255 u = clamp(u, 0.0f, 1.0f);
256 v = clamp(v, 0.0f, 1.0f);
257
258 float sizeS = (float)(width() - 1);
259 float s = u * sizeS;
260 float s0 = std::max(std::floor(s), 0.0f);
261 float s1 = std::min(s0 + 1.0f, sizeS);
262 float smix = s0 < s1 ? (s - s0) / (s1 - s0) : 0.0f;
263
264 float sizeT = (float)(height() - 1);
265 float t = v * sizeT;
266 float t0 = std::max(std::floor(t), 0.0f);
267 float t1 = std::min(t0 + 1.0f, sizeT);
268 float tmix = t0 < t1 ? (t - t0) / (t1 - t0) : 0.0f;
269
270 Pixel UL, UR, LL, LR;
271 read(UL, (unsigned)s0, (unsigned)t0, r);
272 read(UR, (unsigned)s1, (unsigned)t0, r);
273 read(LL, (unsigned)s0, (unsigned)t1, r);
274 read(LR, (unsigned)s1, (unsigned)t1, r);
275
276 Pixel TOP = UL * (1.0f - smix) + UR * smix;
277 Pixel BOT = LL * (1.0f - smix) + LR * smix;
278 pixel = TOP * (1.0f - tmix) + BOT * tmix;
279 }
