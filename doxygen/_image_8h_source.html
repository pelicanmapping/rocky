241  void Image::read_bilinear(Pixel& pixel, float u, float v, unsigned layer) const
242  {
243  u = clamp(u, 0.0f, 1.0f);
244  v = clamp(v, 0.0f, 1.0f);
245 
246  float sizeS = (float)(width() - 1);
247  float s = u * sizeS;
248  float s0 = std::max(std::floor(s), 0.0f);
249  float s1 = std::min(s0 + 1.0f, sizeS);
250  float smix = s0 < s1 ? (s - s0) / (s1 - s0) : 0.0f;
251 
252  float sizeT = (float)(height() - 1);
253  float t = v * sizeT;
254  float t0 = std::max(std::floor(t), 0.0f);
255  float t1 = std::min(t0 + 1.0f, sizeT);
256  float tmix = t0 < t1 ? (t - t0) / (t1 - t0) : 0.0f;
257 
258  Pixel UL, UR, LL, LR;
259  read(UL, (unsigned)s0, (unsigned)t0, layer);
260  read(UR, (unsigned)s1, (unsigned)t0, layer);
261  read(LL, (unsigned)s0, (unsigned)t1, layer);
262  read(LR, (unsigned)s1, (unsigned)t1, layer);
263 
264  Pixel TOP = UL * (1.0f - smix) + UR * smix;
265  Pixel BOT = LL * (1.0f - smix) + LR * smix;
266  pixel = TOP * (1.0f - tmix) + BOT * tmix;
267  }
