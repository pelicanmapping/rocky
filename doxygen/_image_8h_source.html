251 void Image::read_bilinear(Pixel& pixel, float u, float v, unsigned layer) const
252 {
253 u = util::clamp(u, 0.0f, 1.0f);
254 v = util::clamp(v, 0.0f, 1.0f);
255
256 float sizeS = (float)(width() - 1);
257 float s = u * sizeS;
258 float s0 = std::max(std::floor(s), 0.0f);
259 float s1 = std::min(s0 + 1.0f, sizeS);
260 float smix = s0 < s1 ? (s - s0) / (s1 - s0) : 0.0f;
261
262 float sizeT = (float)(height() - 1);
263 float t = v * sizeT;
264 float t0 = std::max(std::floor(t), 0.0f);
265 float t1 = std::min(t0 + 1.0f, sizeT);
266 float tmix = t0 < t1 ? (t - t0) / (t1 - t0) : 0.0f;
267
268 Pixel UL, UR, LL, LR;
269 read(UL, (unsigned)s0, (unsigned)t0, layer);
270 read(UR, (unsigned)s1, (unsigned)t0, layer);
271 read(LL, (unsigned)s0, (unsigned)t1, layer);
272 read(LR, (unsigned)s1, (unsigned)t1, layer);
273
274 Pixel TOP = UL * (1.0f - smix) + UR * smix;
275 Pixel BOT = LL * (1.0f - smix) + LR * smix;
276 pixel = TOP * (1.0f - tmix) + BOT * tmix;
277 }

354 inline constexpr float linear_to_sRGB(float c)
355 {
356 constexpr float cutoff = 0.04045f / 12.92f;
357 constexpr float linearFactor = 12.92f;
358 constexpr float nonlinearFactor = 1.055f;
359 constexpr float exponent = 1.0f / 2.4f;
360 if (c <= cutoff)
361 return c * linearFactor;
362 else
363 return std::pow(c, exponent) * nonlinearFactor - 0.055f;
364 }

367 {
368 constexpr float cutoff = 0.04045f / 12.92f;
369 constexpr float linearFactor = 12.92f;
370 constexpr float nonlinearFactor = 1.055f;
371 constexpr float exponent = 2.4f;
372 if (c <= linearFactor * cutoff)
373 return c / linearFactor;
374 else
375 return std::pow((c + 0.055f) / nonlinearFactor, exponent);
