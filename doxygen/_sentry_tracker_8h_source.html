66 inline void* use(const T& data, void* token)
67 {
68 // Find the tracker for this tile and update its timestamp
69 if (token)
70 {
71 Token* ptr = static_cast<Token*>(token);
72
73 // Move the tracker to the front of the list (ahead of the sentry).
74 // Once a cull traversal is complete, all visited tiles will be
75 // in front of the sentry, leaving all non-visited tiles behind it.
76 _list.splice(_list.begin(), _list, *ptr);
77 *ptr = _list.begin();
78 return ptr;
79 }
80 else
81 {
82 // New entry:
83 Token* ptr = new Token();
84 _list.emplace_front(data, ptr); // ListEntry
85 *ptr = _list.begin();
86 return ptr;
87 }

90 template<typename CALLABLE>
91 inline void flush(unsigned maxCount, CALLABLE&& dispose)
92 {
93 // After cull, all visited tiles are in front of the sentry, and all
94 // non-visited tiles are behind it. Start at the sentry position and
95 // iterate over the non-visited tiles, checking them for deletion.
96 ListIterator i = _sentryptr;
97 ListIterator tmp;
98 unsigned count = 0;
99
100 for (++i; i != _list.end() && count < maxCount; ++i)
101 {
102 ListEntry& le = *i;
103
104 // user disposal function
105 bool disposed = dispose(le._data);
106
107 if (disposed)
108 {
109 // back up the iterator so we can safely erase the entry:
110 tmp = i;
111 --i;
112
113 // delete the token
114 delete static_cast<Token*>(le._token);
115
116 // remove it from the tracker list:
117 _list.erase(tmp);
118 ++count;
119 }
120 }
121
122 // reset the sentry.
123 _list.splice(_list.begin(), _list, _sentryptr);
124 _sentryptr = _list.begin();
125 }
126 };
